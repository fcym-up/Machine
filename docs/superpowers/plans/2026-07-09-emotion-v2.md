# 情绪感知 v2 实施计划

> **面向 agent 执行者：** 必须使用子技能 superpowers:subagent-driven-development（推荐）或 superpowers:executing-plans 来逐任务实施。步骤使用 checkbox（`- [ ]`）语法追踪。

**目标：** 将情绪引擎从「被动调用、单信号源」升级为「主动定时计算 + 高权重信号即时触发」的多信号加权持久化模型。

**架构：** 五个信号源（对话 0.35、音乐 0.25、窗口 0.20、空闲 0.10、时间 0.10）汇入 EmotionComputer，在 1 小时滑动窗口内加权评分并施加时间衰减。结果存入 `emotion_states` 表。对话中的 LLM 共分类以零额外延迟提供最高权重信号。

**技术栈：** Python 3.13, FastAPI, SQLAlchemy 2.x, PostgreSQL 18, DeepSeek API

## 全局约束

- 所有已有 API 保持向后兼容
- LLM 共分类不得增加聊天延迟（复用已有调用）
- 权重可通过数据库动态调整，无需重启
- 不引入新的外部依赖
- 遵循项目现有模式：Base 模型 + `mapped_column`、SessionLocal 会话

    source: Mapped[str] = mapped_column(String(32), nullable=False, index=True)
    emotion_label: Mapped[str] = mapped_column(String(16), nullable=False)

```python
# app/algorithms/emotion_computer.py
"""多信号加权情绪计算引擎。"""
from collections import defaultdict
from datetime import datetime, timezone, timedelta

from sqlalchemy.orm import Session

from app.models.emotion import EmotionSignal, EmotionState, SignalWeight

WINDOW_SECONDS = 3600  # 1 小时滑动窗口
EMOTION_LABELS = ["焦虑", "开心", "疲惫", "专注", "放松", "平静", "沮丧", "好奇"]


class EmotionComputer:

    def load_weights(self, db: Session) -> dict[str, float]:
        rows = db.query(SignalWeight).filter(SignalWeight.enabled.is_(True)).all()
        return {r.source: float(r.weight) for r in rows}

    def compute(self, db: Session) -> EmotionState | None:
        now = datetime.now(timezone.utc)
        cutoff = now - timedelta(seconds=WINDOW_SECONDS)
        signals = db.query(EmotionSignal).filter(
            EmotionSignal.created_at >= cutoff
        ).all()

        if not signals:
            return None

        weights = self.load_weights(db)
        scores: dict[str, float] = defaultdict(float)
        factors: list[str] = []

        for sig in signals:
            source_weight = weights.get(sig.source, 0.1)
            age = (now - sig.created_at).total_seconds()
            decay = max(0.0, 1.0 - age / WINDOW_SECONDS)
            score = source_weight * float(sig.confidence) * decay
            scores[sig.emotion_label] += score
            if sig.payload and sig.payload.get("factor"):
                factors.append(sig.payload["factor"])

        if not scores:
            return None

        sorted_scores = sorted(scores.items(), key=lambda x: -x[1])
        primary = sorted_scores[0][0]
        secondary = sorted_scores[1][0] if len(sorted_scores) > 1 else None
        top_score = sorted_scores[0][1]
        confidence = min(0.95, top_score)
        intensity = min(0.95, max(0.1, top_score * 1.2))

        state = EmotionState(
            primary_emotion=primary,
            secondary_emotion=secondary,
            intensity=round(intensity, 2),
            confidence=round(confidence, 2),
            scores=dict(sorted_scores),
            factors=list(dict.fromkeys(factors))[-5:],
            signal_count=len(signals),
        )
        db.add(state)
        db.commit()
        db.refresh(state)
        return state

    def get_current(self, db: Session) -> EmotionState | None:
        return db.query(EmotionState).order_by(
            EmotionState.created_at.desc()
        ).first()
```

```python
# app/services/emotion_collector.py
"""信号采集器 — 从所有来源收集情绪信号。"""
from datetime import datetime, timezone, timedelta
from sqlalchemy.orm import Session
from loguru import logger

from app.models.emotion import EmotionSignal, SignalWeight
from app.models.event import Event

# 窗口活动类别 → 情绪映射
WINDOW_EMOTION_MAP = {
    "coding": "专注", "browsing": "平静", "social": "平静",
    "gaming": "放松", "music": "放松", "document": "专注",
    "terminal": "专注", "job": "焦虑", "video": "放松",
    "other": "平静",
}

# 小时区间 → 情绪映射
TIME_EMOTION_MAP = {
    (0, 5): "疲惫", (5, 7): "专注", (7, 9): "开心",
    (9, 18): "专注", (18, 21): "放松", (21, 24): "平静",
}


def _get_weight(db: Session, source: str) -> float:
    row = db.query(SignalWeight).filter_by(source=source).first()
    return float(row.weight) if row else 0.1


def collect_window_signals(db: Session) -> int:
    """从最近窗口切换事件中采集信号。"""
    cutoff = datetime.now(timezone.utc) - timedelta(hours=1)
    events = db.query(Event).filter(
        Event.event_type == "user-action",
        Event.source == "activity",
        Event.created_at >= cutoff,
    ).all()

    count = 0
    weight = _get_weight(db, "window")
    for e in events:
        payload = e.payload if isinstance(e.payload, dict) else {}
        category = payload.get("category", "other")
        emotion = WINDOW_EMOTION_MAP.get(category, "平静")
        duration = payload.get("duration", 0)
        confidence = min(0.8, max(0.1, duration / 300))  # 5 分钟封顶

        db.add(EmotionSignal(
            source="window", emotion_label=emotion,
            weight=weight, confidence=confidence,
            payload={
                "category": category,
                "app": payload.get("app", ""),
                "process": payload.get("process", ""),
                "factor": f"在 {payload.get('app','某应用')} 工作了 {duration} 秒",
            },
            event_id=e.id,
        ))
        count += 1
    if count > 0:
        db.commit()
    return count


def collect_idle_signals(db: Session) -> int:
    """采集空闲信号 — 仅在空闲状态变化时发出。"""
    cutoff = datetime.now(timezone.utc) - timedelta(hours=1)
    idle_events = db.query(Event).filter(
        Event.event_type == "system-event",
        Event.source == "system",
        Event.created_at >= cutoff,
    ).all()

    count = 0
    weight = _get_weight(db, "idle")
    for e in idle_events:
        payload = e.payload if isinstance(e.payload, dict) else {}
        if payload.get("state") == "away":
            db.add(EmotionSignal(
                source="idle", emotion_label="疲惫",
                weight=weight, confidence=0.6,
                payload={"factor": "长时间离开键盘"},
                event_id=e.id,
            ))
            count += 1
    if count > 0:
        db.commit()
    return count


def collect_time_signal(db: Session) -> int:
    """采集基于时间的信号（每次计算周期调用一次）。"""
    hour = datetime.now(timezone.utc).hour + 8  # UTC+8
    if hour >= 24:
        hour -= 24

    emotion = "平静"
    for (start, end), em in TIME_EMOTION_MAP.items():
        if start <= hour < end:
            emotion = em
            break

    weight = _get_weight(db, "time")
    db.add(EmotionSignal(
        source="time", emotion_label=emotion,
        weight=weight, confidence=0.5,
        payload={"factor": f"当前时段: {hour}:00"},
    ))
    db.commit()
    return 1


def inject_conversation_signal(
    db: Session, emotion_label: str, confidence: float = 0.7,
    payload: dict | None = None,
) -> EmotionSignal | None:
    """注入对话导出的情绪信号（从聊天 API 调用）。"""
    if emotion_label not in ("焦虑", "开心", "疲惫", "专注", "放松", "平静", "沮丧", "好奇"):
        return None
    weight = _get_weight(db, "conversation")
    sig = EmotionSignal(
        source="conversation", emotion_label=emotion_label,
        weight=weight, confidence=confidence,
        payload=payload or {},
    )
    db.add(sig)
    db.commit()
    db.refresh(sig)
    return sig


def collect_all(db: Session) -> dict:
    """运行所有采集器，返回各来源计数。"""
    result = {}
    try:
        result["window"] = collect_window_signals(db)
    except Exception as e:
        logger.warning(f"窗口信号采集错误: {e}")
        result["window"] = 0
    try:
        result["idle"] = collect_idle_signals(db)
    except Exception as e:
        logger.warning(f"空闲信号采集错误: {e}")
        result["idle"] = 0
    try:
        result["time"] = collect_time_signal(db)
    except Exception as e:
        logger.warning(f"时间信号采集错误: {e}")
        result["time"] = 0
    return result
```

        "scores": state.scores,
        "factors": state.factors or [],
        "signal_count": state.signal_count,

@router.get("/history")
def history(limit: int = Query(24, ge=1, le=168), db: Session = Depends(get_db)):
    states = db.query(EmotionState).order_by(
        EmotionState.created_at.desc()
    ).limit(limit).all()
    return {
        "count": len(states),
        "items": [{
            "id": s.id,
            "primary_emotion": s.primary_emotion,
            "secondary_emotion": s.secondary_emotion,
            "intensity": s.intensity,
            "confidence": s.confidence,
            "created_at": str(s.created_at),
        } for s in states],
    }


@router.get("/weights")
def list_weights(db: Session = Depends(get_db)):
    rows = db.query(SignalWeight).all()
    return {
        "items": [{
            "source": r.source, "weight": float(r.weight),
            "enabled": r.enabled, "updated_at": str(r.updated_at),
        } for r in rows],
    }


@router.put("/weights/{source}")
def update_weight(source: str, data: WeightUpdate, db: Session = Depends(get_db)):
    row = db.query(SignalWeight).filter_by(source=source).first()
    if not row:
        row = SignalWeight(source=source, weight=data.weight, enabled=data.enabled)
        db.add(row)
    else:
        row.weight = data.weight
        row.enabled = data.enabled
    from datetime import datetime, timezone
    row.updated_at = datetime.now(timezone.utc)
    db.commit()
    return {"source": source, "weight": float(row.weight), "enabled": row.enabled}


@router.post("/recompute")
def recompute(db: Session = Depends(get_db)):
    """手动触发重算。"""
    from app.services.emotion_collector import collect_all
    collect_all(db)
    state = compute_current(db)
    if state:
        return {"primary_emotion": state.primary_emotion, "confidence": state.confidence}
    return {"message": "没有可计算的信号"}

# tests/test_emotion_v2.py
"""情绪 v2 系统集成测试。"""
import pytest

from app.models.emotion import EmotionSignal, EmotionState, SignalWeight

    def setup_method(self):
        self.db = SessionLocal()
        # 种子权重
        for src, w in [("conversation", 0.35), ("window", 0.20), ("time", 0.10)]:
            sw = self.db.query(SignalWeight).filter_by(source=src).first()
